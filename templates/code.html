class Solution:
    def trap(self, height: List[int]) -> int:
        '''
        This solution uses a two-pointer approach, with a pointer starting on either side of the list. We will also keep 
        track of the maximum height encountered on either side as we iterate through the list, as this is crucial in 
        determining how much water can be trapped. As we iterate using the two pointers, we will move in the pointer for 
        whatever side has the smaller maximum height encountered. We don't care about the other side's precise height - 
        it is enough to know that it is larger, since that means our water level will surely be bounded by the current 
        maximum height on this side. We keep iterating in this manner, moving in with whatever side has the smaller 
        maximum height and updating values as we progress. This is the basic premise of this algorithm, please refer to 
        the comments below for more details.
        '''
        
        # initialize pointers
        left = 0
        right = len(height) - 1
        
        # initialize maximum heights
        maxLeft = height[left]
        maxRight = height[right]
        
        water = 0
        
        # iterate through list until pointers meet
        while left < right:
            # if maxLeft is less than maxRight, move in with left pointer - we know water must be bounded by maxLeft
            if maxLeft < maxRight:
                left += 1
                
                # see if any water is trapped at current index by maxLeft
                if height[left] < maxLeft:
                    water += (maxLeft - height[left])
                else:
                    maxLeft = height[left]
            # otherwise, move in with right pointer - we know water must be bounded by maxRight
            else:
                right -= 1
                
                # see if any water is trapped at current index by maxRight
                if height[right] < maxRight:
                    water += (maxRight - height[right])
                else:
                    maxRight = height[right]
                    
        return water
